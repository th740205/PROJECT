@app.route("/api/chat", methods=["POST"])
def chat():
    data = request.get_json()  # React에서 보낸 JSON 받기
    message = data.get("message")  # message 키 값 꺼내기

    print("React가 보낸 메시지:", message)

    return jsonify({"reply": f"너가 보낸: {message}"})
